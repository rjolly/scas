import edu.jas.poly.TermOrderByName
import jas.PowerProduct
import scas.base.BigInteger
import BigInteger.int2bigInt

val m = PowerProduct("x")(TermOrderByName.LEX)
val Array(x) = m.generators
import m.given

assert(x > 1)
assert(1 < x)
assert(1 | x)
assert(x * 1 >< 1 * x)
assert(x * x >< x \ 2)
