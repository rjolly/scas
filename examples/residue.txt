import scas.base.{BigInteger, Rational}
import scas.polynomial.tree.UnivariatePolynomial
import scas.residue.Residue
import BigInteger.given
import Rational.given

val r = UnivariatePolynomial(Rational)("x")
val List(x) = r.generators
import r.given

val s = Residue(Rational(x))(x\2-2)
import s.given

assert (x\2 >< 2)
assert (x\2 >< Rational("2"))
assert (s == Rational(x)(x\2-2))
