import scas.power.Lexicographic
import scas.variable.Variable

val m = Lexicographic(0)((for (i <- 0 until 4; j <- 0 until 2) yield Variable("a", Array(i, j): _*)): _*)
val a = m.generators.grouped(2).toList
val s = (for (i <- 0 until 4; j <- 0 until 2) yield a(i)(j)).toList
import m.given

assert(s == List(a(0)(0), a(0)(1), a(1)(0), a(1)(1), a(2)(0), a(2)(1), a(3)(0), a(3)(1)))
