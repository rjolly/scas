// Requires : org.apache.commons#commons-math3;3.6.1

import math3.Matrix
import scas.base.BigInteger
import BigInteger.given

val m = Matrix(3)
import m.given

val a = m(
  0, 1, 0,
  0, 0, 1,
  1, 0, 0
)
val b = m(
  0, 0, 1,
  1, 0, 0,
  0, 1, 0
)
assert(a + 1 >< 1 + a)
assert(a%* 2 >< 2 *%a)
assert(a%/ 2 >< .5*%a)
assert(a \ -1 >< b)
assert(a * b >< 1)
