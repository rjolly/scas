import scas.base.{BigInteger, Rational}
import scas.residue.AlgebraicNumber
import BigInteger.given
import Rational.given

val r = AlgebraicNumber(Rational)("x")
val List(x) = r.generators.toList
import r.given
r.update(2-x\2)

assert (2 >< x\2)
assert (Rational("2") - x\2 >< 0)
assert (r.toString == Rational(x)(2-x\2).toString)
assert (r.math == Rational(x)(2-x\2).math)
