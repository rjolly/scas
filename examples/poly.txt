import scas.rendering.MathObject
import scas.base.BigInteger
import scas.power.Lexicographic
import scas.polynomial.TreePolynomial
import BigInteger.int2bigInt
import scas.{coef2poly, identity}

given r: TreePolynomial[BigInteger, Array[Int]](using BigInteger, Lexicographic[Int]("x", "y", "z"))
val List(x, y, z) = r.generators.toList

assert(x + 1 >< 1 + x)
assert(x + BigInteger("1") >< BigInteger("1") + x)

given s: TreePolynomial[TreePolynomial.Element[BigInteger, Array[Int]], Array[Int]](using BigInteger(x, y, z), Lexicographic[Int]("a"))
val List(a) = s.generators.toList

assert(a + 1 >< 1 + a)
assert(a + BigInteger("1") >< BigInteger("1") + a)
assert(a + x >< x+a)
assert((a + x).show == "x+a")
assert((a + x).math == s.math(x+a))
assert(MathObject(a + x).toString == "x+a")
